  • 點我展開
    genHTMLTable <- function(OpeningHourVector) {

		# 首先確定輸入值，進行基本的判斷，如果不符合規範直接 return()
    if (is.na(OpeningHourVector) | 
        !is.character((OpeningHourVector)) | 
        nchar(OpeningHourVector) != 21
    ) { return() }
    str <- OpeningHourVector

		# 將 長達 21 的字串先以 7, 7, 7 的長度切成三小塊
    s <- sapply(
        seq(
            from=1, 
            to=nchar(str), 
            by=7
        ), 
        function(i) substr(str, i, i+6)
    )

		# 用 R Shiny 的 HTML 元素產生表格
    tags$table(
        style = "width: 100%;",

				# 表格header
        tags$tr(
            tags$th(""),
            tags$th("一"),
            tags$th("二"),
            tags$th("三"),
            tags$th("四"),
            tags$th("五"),
            tags$th("六"),
            tags$th("日")
        ),

				# 用 lapply 代替 for
        lapply(X = 1:3, FUN = function(i) {
            tags$tr(
                lapply(X = 0:7, FUN = function(j) {

										# 將字串一個字元一個字元切開放到 ss 裡面
                    ss <- strsplit(s[i], "")
                    
                    tags$td(

												# 如果 i = 1:3 且 j = 0 時，產生第一行元素，早上/下午/晚上
                        if (j == 0) {
                            if (i == 1 & j == 0) { "上午" }
                            else if (i == 2 & j == 0) { "下午" }
                            else if (i == 3 & j == 0) { "晚上" }
                        }  else {

														# 如果資料是 "N" 印出勾勾，表示有營業；如果是"Y"，表示休息，vice versa
                            if (ss[[1]][j] == "N") { tags$i(class = "fas fa-check-circle", style = "color: mediumaquamarine;") } 
                            else { tags$i(class = "fas fa-times-circle", style = "color: tomato;") }
                        },
                        style = "padding-top: 4px; padding-bottom: 4px;"
                    )
                })
            )
        })
    )
}

說明

用GPS定位當前位置與附近的幾家藥局做時間預估路線規劃

◾透過瀏覽器要求GPS權限取得使用者位置後，在拿來與資料庫中的藥局位置比較。

程式碼說明

# observe when user is Located, save coordinates.
observe({

		# 當使用者使用 leaflet.extras 的 gpsControl 定位時，監測到gps打開，
		# map_gps_located變數發生變化，透過 reactive values rv 設定變數
    x <- input$map_gps_located
    if (!is.null(x)) { rv$isLocated <- TRUE }

		# 並儲存位置
    rv$gpsLat <- x$coordinates[1]
    rv$gpsLng <- x$coordinates[2]
})

接著，在 Pop up 視窗產生時判斷使用者有沒有定位，生出路線規劃按鈕。

結論

這題的答案比較難找，需要自己去翻 leaflet.extras 的原始碼才找的到使用者已經定位會觸發什麼。
